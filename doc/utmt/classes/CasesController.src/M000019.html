  def show
    # TODO: Fiksumpi tapa hoitaa tämä to_json metodein.
    # Nyt jostain syystää tulee CircularReferenceError, jos käyttää
    # @case.to_json
    @case = Case.find(params[:id])
    ret = '{"data":[{"id":' + @case.id.to_json + ', "title":' + @case.title.to_json + ', "time_estimate":' + @case.time_estimate.to_json + ', "created_by":' + @case.creator.login.to_json + ', "created_at":"' + @case.created_at.strftime("%Y-%m-%d %H:%M:%S") + '", "updated_by":' + @case.updater.login.to_json + ', "updated_at":"' + @case.updated_at.strftime("%Y-%m-%d %H:%M:%S") + '", "objective":' + @case.objective.to_json + ', "test_data":' + @case.test_data.to_json + ', "preconditions_and_assumptions":' + @case.preconditions_and_assumptions.to_json + ', "steps":'+@case.steps.to_json(:only => [:id,:action,:result,:order], :methods => [:history]) + '}]}'
    render :json => ret
  end
