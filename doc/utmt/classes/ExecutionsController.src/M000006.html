  def destroy
    @exec = Execution.find(params[:id])
    if (@exec.destroy)
      render :json => hash_to_extjson({'id' => params[:id]})
      #render :nothing => true, :status => 204
      #render :json => "{\"status\":#{ok}}"
    else
      # Exception handling
    end
  end
