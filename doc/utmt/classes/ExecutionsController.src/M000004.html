  def show
    @execution = Execution.find( params[:id])

    # Merge attributes from associated object(s).
    # Active Record .attributes() doesn't give us values 
    # from has_many or has_and_belongs_to_many associations.
    attributes = Hash[ 'assigned_user_ids', @execution.assigned_user_ids]
    attributes.merge!( @execution.attributes)

    render :json => hash_to_extjson( attributes)
  end
