  def executions_tree
    ret = '['
    comma = ''
    @current_project.executions.each{ |e|
      ret += comma + '{"text":"' + e.test_set.name + ': ' + e.name + 
        '", "id":"' + e.id.to_s + '", "leaf":true}'
      comma = ', '  
    }
    ret += ']'
    render :json => ret
  end
