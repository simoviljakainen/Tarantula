  def show
    ret = '{ }'
    if (params[:id])
      @set = TestSet.find(params[:id])
      ret = '{"data":[{"name":' + @set.name.to_json + ', "id":' + @set.id.to_json + ',  "cases": ' + @set.cases_test_sets.collect{|s|s.test_case}.to_treejson(:title) + '}]}'
    end
    render :json => ret
  end
