  def index
    sets = []
    # Should we return all test sets, if project is not specified?
    if (params[:project_id])
      # Listaa aktiivisen projektin setit JSON-muodossa joka voidaan sijoittaa
      # Ext.js:n puukomponentin sisällöksi.
      sets = TestSet.find(:all, :conditions => ['`project_id` =?', params[:project_id]])
    end
    if (params[:format] == 'tree')
      render :json => sets.to_treejson(:name)
    else
      render :json => array_to_extjson( sets)    
    end
  end
