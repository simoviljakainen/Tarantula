  def executions_tree
    ret = '['
    comma = ''
    @current_user.executions.each {|e|
      if ((e.test_set.cases.count > 0) and (e.result == 'NOT_RUN') and (e.project_id == session[:project]))
        ret += comma + '{"text":"' + e.test_set.name + ': ' + e.name + '", "id":"exec' + e.id.to_s + '", "leaf":true}'
        comma = ', '
      end
    }
    render :json => ret + ']'
  end
