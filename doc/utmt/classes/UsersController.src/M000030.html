  def show
    if (params[:id] == @current_user.id)
      render :json => @current_user.to_json(:only => [:description, :phone, :realname, :login, :email], :methods => [:time_zone])
    else
      @user = User.find(params[:id])
      render :json => "{\"data\":[#{@user.to_json(:only => [:id, :login, :realname, :email, :phone, :admin])}]}"
    end
  end
