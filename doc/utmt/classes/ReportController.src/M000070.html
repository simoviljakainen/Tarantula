  def test_set_duration
    table = Execution.report_table(:all, :conditions => ["executions.test_set_id =? AND executions.`created_at` >= ? AND executions.`created_at` <= ?", params[:id], params[:sdate], params[:edate]], :only => [:name],
              :methods => [:run_by, :duration, :start_date, :end_date, :number_of_cases, :skipped_cases],
      :transforms => lambda {|r|
        r["start_date"] = r["start_date"].strftime("%Y-%m-%d %H:%M")
        r["end_date"] = r["end_date"].strftime("%Y-%m-%d %H:%M")
      }
    )
    #table.replace_column("start_date", "Start date") {|r| puts r}
    #table.replace_column("end_date", "End date") {|r| r.strformat("%Y-%m-%d")}
    render :text => table.to_html
