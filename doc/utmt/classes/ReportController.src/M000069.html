  def case_execute_duration
    table = TestSet.report_table(params[:sets], :only => [],
              :include => {
                :cases => {
                  :only => [:title],
                  :methods => [:avg_duration, :median_duration, :passed_count, :failed_count, :skipped_count]
                }
              })
    render :text => table.to_html
  end
